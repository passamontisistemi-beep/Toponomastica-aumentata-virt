function fetchWikipediaNearby(lat, lon) { const radius = 1500; // metri (aumentato) const url = 'https://it.wikipedia.org/w/api.php?' + 'action=query&list=geosearch&gscoord=' + lat + '%7C' + lon + '&gsradius=' + radius + '&gslimit=20&format=json&origin=*'; console.log("Richiesta Wikipedia:", url); fetch(url) .then(res => res.json()) .then(data => { console.log("Risposta Wikipedia:", data); if (!data.query || !data.query.geosearch) { statusText.textContent = 'Nessun punto di interesse trovato nelle vicinanze.'; return; } const results = data.query.geosearch; if (results.length === 0) { statusText.textContent = 'Nessun punto di interesse trovato entro 1.5 km.'; return; } results.forEach((poi, index) => { addPoiMarker(poi, index + 1); }); }) .catch(err => { console.error("Errore Wikipedia:", err); statusText.textContent = 'Errore nel recupero dei punti di interesse.'; }); }
